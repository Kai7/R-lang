fpermu <- replicate(1000, {
  all <- sample(c(x_norm,x_hf))
  new_x_norm <- all[1:n]
  new_x_hf <- all[(n+1):(2*n)]
  return(var(new_x_norm)/var(new_x_hf))
})
round((sum(fpermu < myF) + 1) / (length(fpermu) + 1), 4)
